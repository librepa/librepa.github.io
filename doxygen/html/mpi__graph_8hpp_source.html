89 inline boost::mpi::communicator
90 directed_graph_communicator(MPI_Comm parent_communicator,
91  const std::vector<int> &source_ranks,
92  const std::vector<int> &destination_ranks)
93 {
94  MPI_Comm comm;
95  // Edges to all processes in "neighbors"
96  MPI_Dist_graph_create_adjacent(
97  parent_communicator, source_ranks.size(), source_ranks.data(),
98  static_cast<const int *>(MPI_UNWEIGHTED), destination_ranks.size(),
99  destination_ranks.data(), static_cast<const int *>(MPI_UNWEIGHTED),
100  MPI_INFO_NULL, 0, &comm);
101 
102 #ifndef NDEBUG
103  int indegree = 0, outdegree = 0, weighted = 0;
104  MPI_Dist_graph_neighbors_count(comm, &indegree, &outdegree, &weighted);
105  assert(!weighted);
106  assert(static_cast<size_t>(indegree) == source_ranks.size());
107  assert(static_cast<size_t>(outdegree) == destination_ranks.size());
108  std::vector<int> __ineighs(indegree, -1), __iw(indegree, -1);
109  std::vector<int> __oneighs(outdegree, -1), __ow(outdegree, -1);
110  MPI_Dist_graph_neighbors(comm, indegree, __ineighs.data(), __iw.data(),
111  outdegree, __oneighs.data(), __ow.data());
112  for (size_t i = 0; i < source_ranks.size(); ++i)
113  assert(__ineighs[i] == source_ranks[i]);
114  for (size_t i = 0; i < destination_ranks.size(); ++i)
115  assert(__oneighs[i] == destination_ranks[i]);
116 #endif
117 
118  return boost::mpi::communicator{comm, boost::mpi::comm_take_ownership};
119 }
