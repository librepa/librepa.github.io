121  GlobalBox(Vec3d box_size, double min_cell_size)
122  : m_cell_grid(static_cast<index_type_3d>(box_size[0] / min_cell_size),
123  static_cast<index_type_3d>(box_size[1] / min_cell_size),
124  static_cast<index_type_3d>(box_size[2] / min_cell_size)),
125  m_cell_size(box_size[0] / m_cell_grid[0],
126  box_size[1] / m_cell_grid[1],
127  box_size[2] / m_cell_grid[2])
128  {
129  constexpr auto zero = index_type_3d{0};
130  m_cell_grid_corr[0]
131  = linearize(cell_index_type{m_cell_grid[0], zero, zero});
132  m_cell_grid_corr[1]
133  = linearize(cell_index_type{zero, m_cell_grid[1], zero});
134  m_cell_grid_corr[2]
135  = linearize(cell_index_type{zero, zero, m_cell_grid[2]});

137  std::transform(std::begin(util::NeighborOffsets3D::raw),
138  std::end(util::NeighborOffsets3D::raw),
139  std::begin(m_neigh_offset_1d), [this](const auto &cell) {
140  return this->linearize(cell);
141  // ^^^^^^ Note: gcc-5.x compatibility
142  });
143  }

149  inline index_type_1d neighbor(index_type_1d index, fs_neighidx neigh) const
150  {
151  auto idx = unlinearize(index);
152  const auto &no = util::NeighborOffsets3D::raw[neigh];
153  auto ni = index + m_neigh_offset_1d[neigh];
154 
155  for (int d = 0; d < 3; ++d) {
156  // Can be out of bounds by at most 1, i.e. subtracting m_cell_grid
157  // once suffices.
158  if (idx[d] == 0 && no[d] < 0)
159  ni += index_type_1d{m_cell_grid_corr[d]};
160  else if (idx[d] == m_cell_grid[d] - 1 && no[d] > 0)
161  ni -= index_type_1d{m_cell_grid_corr[d]};
162  }
163  return ni;
164  }
