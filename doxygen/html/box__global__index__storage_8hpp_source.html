62  void init(Vec3i global_grid_size,
63  Vec3i local_grid_size,
64  Vec3i local_lower_left_idx3d)
65  {
66  using namespace util::vector_arithmetic;
67 
68  _global_grid_size = global_grid_size;
69  _local_lower_left_idx3d = local_lower_left_idx3d;
70  _local_grid_size = local_grid_size;
71 
72  // Reserve full halo
73  _ghost_cells.reserve(product(_local_grid_size + 2)
74  - product(_local_grid_size));
75 
76  auto ghost_cell_no = ghost_cell_index_type{0};
77  // Assemble ghost->global and global->ghost mappings -- iterate over
78  // full halo and find canonical representation of cell
79  // Careful: this gets ugly.
80  // Halo are cells that have at least one
81  // coordinate == -1 or == local_grid_size[d].
82  for (int d = 0; d < 3; ++d) {
83  Vec3i local_ghost_idx3d;
84  // The outer most loop has only two iterations
85  // left (-1) and right (_local_grid_size[d]) side of the volume
86  // spanned by _local_grid_size
87  for (local_ghost_idx3d[d] = -1;
88  local_ghost_idx3d[d] <= _local_grid_size[d];
89  local_ghost_idx3d[d] += _local_grid_size[d] + 1) {
90 
91  // Iterate over 2d plane in codimension(d) with coordinate
92  // ghost_idx3d[d] fixed.
93  for (local_ghost_idx3d[_REPA_NDIR(d)] = -1;
94  local_ghost_idx3d[_REPA_NDIR(d)]
95  <= _local_grid_size[_REPA_NDIR(d)];
96  local_ghost_idx3d[_REPA_NDIR(d)]++) {
97 
98  for (local_ghost_idx3d[_REPA_NNDIR(d)] = -1;
99  local_ghost_idx3d[_REPA_NNDIR(d)]
100  <= _local_grid_size[_REPA_NNDIR(d)];
101  local_ghost_idx3d[_REPA_NNDIR(d)]++) {
102 
103  const auto canon_idx3d = get_canonical_vector(
104  _local_lower_left_idx3d + local_ghost_idx3d);
105 
106  // Sort out local cells
107  if (all(canon_idx3d >= _local_lower_left_idx3d)
108  && all(canon_idx3d < (_local_lower_left_idx3d
109  + _local_grid_size)))
110  continue;
111 
112  const auto canon_idx
113  = get_global_cell_index(canon_idx3d);
114 
115  if (_inverse_ghost_map.find(canon_idx)
116  != std::end(_inverse_ghost_map))
117  continue;
118 
119  _inverse_ghost_map.emplace(canon_idx, ghost_cell_no);
120  _ghost_cells.push_back(canon_idx);
121  ghost_cell_no++;
122  }
123  }
124  }
125  }
126 
127  // Free difference between full halo and actual number of ghost cells
128  _ghost_cells.shrink_to_fit();
129  }
