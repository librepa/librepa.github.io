45 struct FlowCalculator {
46  /*
47  * Determines the status of each process (underloaded, overloaded)
48  * in the neighborhood given the local load and returns the volume of load
49  * to send to each neighbor. On underloaded processes, returns a vector of
50  * zeros.
51  *
52  * This call is collective on neighcomm.
53  *
54  * @param neighcomm Graph communicator which reflects the neighbor
55  * relationship amongst processes (undirected edges), without edges to the
56  * process itself.
57  * @param neighbors The ranks of the neighbors of the calling process.
58  * @param load The load of the calling process.
59  * @returns Vector of load values ordered according to the neighborhood
60  * ordering in neighcomm.
61  */
62  virtual PerNeighbor<double>
63  compute_flow(boost::mpi::communicator neighcomm,
64  const std::vector<rank_type> &neighbors,
65  double load) const = 0;
66 
67  virtual ~FlowCalculator() = default;
68 };
