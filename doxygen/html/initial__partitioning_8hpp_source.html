110  const auto cellidx_3d = util::unlinearize(g, _grid_size);
111  const Vec3i procidx3d

113  constant_vec3(0), _dims - 1);

154  StaticRankAssigner(InitialPartitionType pt,
155  const GBox &gbox,
156  const boost::mpi::communicator &comm)
157  : _gbox(gbox)
158  {
161  auto impl_factory = [&]() -> impl::StaticRankAssigner_impl<GBox> * {
162  switch (pt) {
163  case InitialPartitionType::LINEAR:
164  return new impl::Linear_StaticRankAssigner_impl<GBox>(gbox,
165  comm);
166  case InitialPartitionType::CARTESIAN1D:
167  case InitialPartitionType::CARTESIAN2D:
168  case InitialPartitionType::CARTESIAN3D:
169  return new impl::Cart_StaticRankAssigner_impl<GBox>(
170  gbox, make_init_part_communicator(comm, pt));
171  }
172  ensure_not_reached();
173  };
174 
175  _impl = std::unique_ptr<impl::StaticRankAssigner_impl<GBox>>(
176  impl_factory());
177  }
